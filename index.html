Week 1: 作業系統概論

學習目標

  • 了解作業系統的基本概念與功能
  • 認識作業系統的歷史發展
  • 學習作業系統的架構與組成

課程內容

作業系統是管理計算機硬體與軟體資源的系統軟體，提供使用者與硬體之間的介面。主要功能包括行程管理、記憶體管理、檔案系統管理和輸入輸出管理。

程式範例：取得系統資訊

system_info.c
#include <stdio.h>
#include <unistd.h>
#include <sys/utsname.h>

int main() {
    struct utsname sys_info;
    
    if (uname(&sys_info) == 0) {
        printf("系統名稱: %s\n", sys_info.sysname);
        printf("節點名稱: %s\n", sys_info.nodename);
        printf("版本: %s\n", sys_info.release);
        printf("架構: %s\n", sys_info.machine);
    }
    
    return 0;
}

Week 2: 行程概念與系統呼叫

學習目標

  • 理解行程 (Process) 的概念
  • 學習行程的狀態與生命週期

行程狀態

行程在執行過程中會經歷不同的狀態：新建、就緒、執行、等待和終止。

程式範例：建立子行程

fork_example.c
#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>

int main() {
    pid_t pid;
    
    printf("父行程開始，PID: %d\n", getpid());
    
    pid = fork();
    
    if (pid < 0) {
        fprintf(stderr, "建立行程失敗\n");
        return 1;
    }
    else if (pid == 0) {
        printf("子行程執行中，PID: %d，父PID: %d\n", 
               getpid(), getppid());
    }
    else {
        printf("父行程繼續執行，子行程PID: %d\n", pid);
    }
    
    return 0;
}
exec_example.c
#include <stdio.h>
#include <unistd.h>

int main() {
    printf("執行 ls 命令：\n");
    
    execl("/bin/ls", "ls", "-l", NULL);
    
    printf("這行不會被印出\n");
    
    return 0;
}
注意：

fork() 會建立一個與父行程幾乎完全相同的子行程。exec 系列函數則會用新程式取代當前行程的映像。

Week 3: 行程排程

學習目標

  • 了解 CPU 排程的概念與目的

常見排程演算法

  • 先來先服務 (FCFS)
  • 最短工作優先 (SJF)
  • 優先權排程 (Priority)
  • 循環排程 (Round Robin)

程式範例：FCFS 排程模擬

fcfs_scheduling.c
#include <stdio.h>

typedef struct {
    int pid;
    int arrival_time;
    int burst_time;
    int waiting_time;
    int turnaround_time;
} Process;

void calculate_fcfs(Process processes[], int n) {
    int current_time = 0;
    
    for (int i = 0; i < n; i++) {
        if (current_time < processes[i].arrival_time) {
            current_time = processes[i].arrival_time;
        }
        
        processes[i].waiting_time = current_time - processes[i].arrival_time;
        current_time += processes[i].burst_time;
        processes[i].turnaround_time = processes[i].waiting_time + processes[i].burst_time;
    }
}

void print_results(Process processes[], int n) {
    printf("PID\t到達\t爆發\t等待\t周轉\n");
    
    float total_waiting = 0, total_turnaround = 0;
    
    for (int i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\n",
               processes[i].pid,
               processes[i].arrival_time,
               processes[i].burst_time,
               processes[i].waiting_time,
               processes[i].turnaround_time);
        
        total_waiting += processes[i].waiting_time;
        total_turnaround += processes[i].turnaround_time;
    }
    
    printf("\n平均等待時間: %.2f\n", total_waiting / n);
    printf("平均周轉時間: %.2f\n", total_turnaround / n);
}

int main() {
    Process processes[] = {
        {1, 0, 5, 0, 0},
        {2, 1, 3, 0, 0},
        {3, 2, 8, 0, 0},
        {4, 3, 6, 0, 0}
    };
    
    int n = sizeof(processes) / sizeof(processes[0]);
    
    calculate_fcfs(processes, n);
    print_results(processes, n);
    
    return 0;
}

程式範例：建立 POSIX 執行緒

pthread_basic.c
#include <stdio.h>
#include <pthread.h>
#include <unistd.h>

void* thread_function(void* arg) {
    int thread_num = *(int*)arg;
    
    printf("執行緒 %d 開始執行\n", thread_num);
    sleep(1);
    printf("執行緒 %d 結束執行\n", thread_num);
    
    return NULL;
}

int main() {
    pthread_t threads[3];
    int thread_args[3];
    
    for (int i = 0; i < 3; i++) {
        thread_args[i] = i + 1;
        pthread_create(&threads[i], NULL, thread_function, &thread_args[i]);
    }
    
    for (int i = 0; i < 3; i++) {
        pthread_join(threads[i], NULL);
    }
    
    printf("所有執行緒已完成\n");
    
    return 0;
}

程式範例：使用 Mutex

mutex_example.c
#include <stdio.h>
#include <pthread.h>

int shared_counter = 0;
pthread_mutex_t mutex;

void* increment_counter(void* arg) {
    for (int i = 0; i < 100000; i++) {
        pthread_mutex_lock(&mutex);
        shared_counter++;
        pthread_mutex_unlock(&mutex);
    }
    return NULL;
}

int main() {
    pthread_t thread1, thread2;
    
    pthread_mutex_init(&mutex, NULL);
    
    pthread_create(&thread1, NULL, increment_counter, NULL);
    pthread_create(&thread2, NULL, increment_counter, NULL);
    
    pthread_join(thread1, NULL);
    pthread_join(thread2, NULL);
    
    pthread_mutex_destroy(&mutex);
    
    printf("最終計數值: %d\n", shared_counter);
    
    return 0;
}

程式範例：動態記憶體配置

dynamic_memory.c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
    int id;
    char name[50];
    float gpa;
} Student;

int main() {
    int n;
    Student* students;
    
    printf("請輸入學生人數: ");
    scanf("%d", &n);
    
    students = (Student*)malloc(n * sizeof(Student));
    
    if (students == NULL) {
        fprintf(stderr, "記憶體配置失敗\n");
        return 1;
    }
    
    for (int i = 0; i < n; i++) {
        students[i].id = i + 1;
        printf("輸入學生 %d 的姓名: ", i + 1);
        scanf("%s", students[i].name);
        printf("輸入 GPA: ");
        scanf("%f", &students[i].gpa);
    }
    
    printf("\n學生資料：\n");
    for (int i = 0; i < n; i++) {
        printf("ID: %d, 姓名: %s, GPA: %.2f\n",
               students[i].id,
               students[i].name,
               students[i].gpa);
    }
    
    free(students);
    
    return 0;
}
重要：

動態配置的記憶體必須使用 free() 釋放，否則會造成記憶體洩漏。

  • LRU (Least Recently Used)

程式範例：FIFO 分頁置換

fifo_page_replacement.c
#include <stdio.h>
#include <stdbool.h>

#define FRAME_SIZE 3

bool is_in_frames(int frames[], int num_frames, int page) {
    for (int i = 0; i < num_frames; i++) {
        if (frames[i] == page) {
            return true;
        }
    }
    return false;
}

int main() {
    int pages[] = {7, 0, 1, 2, 0, 3, 0, 4, 2, 3, 0, 3, 2};
    int n = sizeof(pages) / sizeof(pages[0]);
    
    int frames[FRAME_SIZE];
    for (int i = 0; i < FRAME_SIZE; i++) {
        frames[i] = -1;
    }
    
    int page_faults = 0;
    int next_frame = 0;
    
    printf("FIFO 分頁置換模擬\n\n");
    
    for (int i = 0; i < n; i++) {
        printf("存取分頁 %d: ", pages[i]);
        
        if (!is_in_frames(frames, FRAME_SIZE, pages[i])) {
            frames[next_frame] = pages[i];
            next_frame = (next_frame + 1) % FRAME_SIZE;
            page_faults++;
            printf("分頁錯誤! ");
        } else {
            printf("命中! ");
        }
        
        printf("框架: [");
        for (int j = 0; j < FRAME_SIZE; j++) {
            if (frames[j] == -1) {
                printf(" - ");
            } else {
                printf(" %d ", frames[j]);
            }
        }
        printf("]\n");
    }
    
    printf("\n總分頁錯誤次數: %d\n", page_faults);
    
    return 0;
}

程式範例：檔案操作

file_operations.c
#include <stdio.h>

void write_file(const char* filename, const char* content) {
    FILE* file = fopen(filename, "w");
    
    if (file == NULL) {
        perror("開啟檔案失敗");
        return;
    }
    
    fprintf(file, "%s", content);
    fclose(file);
}

void read_file(const char* filename) {
    FILE* file = fopen(filename, "r");
    
    if (file == NULL) {
        perror("開啟檔案失敗");
        return;
    }
    
    char buffer[256];
    while (fgets(buffer, sizeof(buffer), file) != NULL) {
        printf("%s", buffer);
    }
    
    fclose(file);
}

int main() {
    write_file("test.txt", "Hello, OS Course!\n");
    read_file("test.txt");
    
    return 0;
}

程式範例：TCP 伺服器

tcp_server.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>

#define PORT 8080

int main() {
    int server_fd, client_fd;
    struct sockaddr_in server_addr, client_addr;
    socklen_t addr_len = sizeof(client_addr);
    char buffer[1024] = {0};
    
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);
    
    bind(server_fd, (struct sockaddr*)&server_addr, sizeof(server_addr));
    listen(server_fd, 3);
    
    printf("伺服器正在監聽 Port %d...\n", PORT);
    
    client_fd = accept(server_fd, (struct sockaddr*)&client_addr, &addr_len);
    read(client_fd, buffer, 1024);
    printf("收到訊息: %s\n", buffer);
    
    send(client_fd, "訊息已收到", 16, 0);
    
    close(client_fd);
    close(server_fd);
    
    return 0;
}

專案架構

mini_os.h
#ifndef MINI_OS_H
#define MINI_OS_H

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

#define MAX_PROCESSES 10
#define MAX_MEMORY_BLOCKS 20
#define MAX_FILES 50

typedef enum {
    READY,
    RUNNING,
    WAITING,
    TERMINATED
} ProcessState;

typedef struct {
    int pid;
    char name[50];
    int priority;
    int burst_time;
    int remaining_time;
    ProcessState state;
} Process;

typedef struct {
    int start;
    int size;
    bool allocated;
    int process_id;
} MemoryBlock;

typedef struct {
    char name[50];
    int size;
    char content[1024];
    bool exists;
} File;

typedef struct {
    Process processes[MAX_PROCESSES];
    int process_count;
    MemoryBlock memory[MAX_MEMORY_BLOCKS];
    int memory_count;
    File files[MAX_FILES];
    int file_count;
} MiniOS;

void init_os(MiniOS* os);
void create_process(MiniOS* os, const char* name, int priority, int burst_time);
void schedule_processes(MiniOS* os);
void list_processes(MiniOS* os);
void list_memory(MiniOS* os);
void list_files(MiniOS* os);

#endif
